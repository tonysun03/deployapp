# 数据结构
## 栈 `stack`
### 实现
```js
function Stack() {
    let items = []
    // 入栈
    this.push = function (element) {
        items.push(element)
    }
    this.pop = function () {
        return items.pop()
    }
    this.peek = function () {
        return items[items.length - 1]
    }
    this.isEmpty = function (){
        return items.length === 0
    }
    this.size = function () {
        return items.length
    }
    this.clear = function () {
        items = []
    }
    this.print =function () {
        console.log(items.toString())
    }
}
```
### 从十进制到二进制
```js
function divideBy2(number) {
    let stack = new Stack()
    let rem
    let result
    while (number > 0){
        rem = Math.floor(number%2)
        stack.push(rem)
        number = (number - rem) / 2
    }
    while (!stack.isEmpty) {
        result += stack.pop().toString()
    }
    return result
}
```
